# What is this ?? 
This is a AI based in python without needing the use of a api key
# Installation 
* Linux

  sudo apt update

  sudo apt upgrade
  
  sudo apt install git
  
  sudo apt install python3-pip
  
  git clone https://github.com/OWPMA/Dipxy-AI
  
  cd Dipxy-AI

  pip3 install requests
  
  python3 dipxyai.py
  
  * Termux
    
    pkg update

   *Windows/android
